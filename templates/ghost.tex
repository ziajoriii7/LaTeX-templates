function get_adjusted_time(hours_offset)
    local current_time = os.time()
    local adjusted_time = os.date("*t", current_time + hours_offset * 3600)
    tex.sprint(string.format("%02d:%02d", adjusted_time.hour, adjusted_time.min))
end
\end{luacode}
